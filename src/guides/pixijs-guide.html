Configuración Inicial

Configuración Básica
// Crear aplicación PIXI
const app = new PIXI.Application({
  width: 800,
  height: 600,
  backgroundColor: 0x1099bb
});

// Agregar al DOM
document.body.appendChild(app.view);
Loop de Animación
// Función de animación
app.ticker.add((delta) => {

// O usando requestAnimationFrame
function animate() {
  requestAnimationFrame(animate);

Comandos Principales

Sprites
// Crear sprite desde textura
const texture = PIXI.Texture.from('bunny.png');
const bunny = new PIXI.Sprite(texture);

// Posicionar sprite
bunny.x = 100;
bunny.y = 100;

// Agregar a la escena
app.stage.addChild(bunny);
Gráficos
// Crear gráficos vectoriales
const graphics = new PIXI.Graphics();

// Dibujar rectángulo
graphics.beginFill(0xff0000);
graphics.drawRect(0, 0, 100, 100);
graphics.endFill();

app.stage.addChild(graphics);
Texto
// Crear texto
const style = new PIXI.TextStyle({
  fontFamily: 'Arial',
  fontSize: 36,
  fill: 'white'
});

const text = new PIXI.Text('Hello PIXI!', style);
app.stage.addChild(text);

Interactividad

Eventos de Mouse
// Hacer sprite interactivo
sprite.interactive = true;
sprite.buttonMode = true;

// Eventos de click
sprite.on('pointerdown', onClick);
sprite.on('pointerover', onHover);

function onClick() {
  console.log('Sprite clicked!');
}
Drag and Drop
// Habilitar arrastre
sprite.interactive = true;
sprite.on('pointerdown', onDragStart);
sprite.on('pointerup', onDragEnd);
sprite.on('pointermove', onDragMove);

function onDragStart(event) {
  this.data = event.data;
  this.dragging = true;
}

Características Avanzadas

Filtros
// Aplicar filtros
const blurFilter = new PIXI.filters.BlurFilter();
blurFilter.blur = 5;

sprite.filters = [blurFilter];

// Filtro de color
const colorMatrix = new PIXI.filters.ColorMatrixFilter();
colorMatrix.sepia();
sprite.filters = [colorMatrix];
Contenedores
// Crear contenedor
const container = new PIXI.Container();

// Agregar sprites al contenedor
container.addChild(sprite1);
container.addChild(sprite2);

// Transformar todo el contenedor
container.x = 100;
container.rotation = 0.1;

app.stage.addChild(container);
