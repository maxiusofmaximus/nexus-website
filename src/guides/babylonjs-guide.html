Configuración Inicial

Configuración Básica
// Crear motor y escena
const canvas = document.getElementById("renderCanvas");
const engine = new BABYLON.Engine(canvas, true);
const scene = new BABYLON.Scene(engine);

// Crear cámara
const camera = new BABYLON.ArcRotateCamera("camera", 0, 0, 10, BABYLON.Vector3.Zero(), scene);
Loop de Renderizado
// Iniciar loop de renderizado
engine.runRenderLoop(() => {
  scene.render();
});

// Redimensionar automáticamente
window.addEventListener("resize", () => {
  engine.resize();
});

Comandos Principales

Crear Meshes
// Crear primitivas
const box = BABYLON.MeshBuilder.CreateBox("box", {size: 2}, scene);
const sphere = BABYLON.MeshBuilder.CreateSphere("sphere", {diameter: 2}, scene);
const ground = BABYLON.MeshBuilder.CreateGround("ground", {width: 6, height: 6}, scene);
Materiales
// Crear material
const material = new BABYLON.StandardMaterial("material", scene);
material.diffuseColor = new BABYLON.Color3(1, 0, 1);
material.specularColor = new BABYLON.Color3(0, 1, 0);
box.material = material;
Iluminación
// Crear luces
const light = new BABYLON.HemisphericLight("light", new BABYLON.Vector3(0, 1, 0), scene);
light.intensity = 0.7;

const dirLight = new BABYLON.DirectionalLight("dirLight", new BABYLON.Vector3(0, -1, 0), scene);

Características Avanzadas

Física
// Habilitar física
scene.enablePhysics(new BABYLON.Vector3(0, -9.81, 0), new BABYLON.CannonJSPlugin());

// Agregar física a un mesh
box.physicsImpostor = new BABYLON.PhysicsImpostor(box, BABYLON.PhysicsImpostor.BoxImpostor, { mass: 1, restitution: 0.7 }, scene);
Animaciones
// Crear animación
const animationBox = new BABYLON.Animation("boxAnimation", "position.x", 30, BABYLON.Animation.ANIMATIONTYPE_FLOAT, BABYLON.Animation.ANIMATIONLOOPMODE_CYCLE);

// Definir keyframes
const keys = [];
keys.push({frame: 0, value: 0});
keys.push({frame: 30, value: 5});
animationBox.setKeys(keys);
