Configuración Inicial

Configuración Básica
// Configuración del juego
const config = {
  type: Phaser.AUTO,
  width: 800,
  height: 600,
  scene: {
    preload: preload,
    create: create,
    update: update
  }
};

const game = new Phaser.Game(config);

  // Lógica del juego

Comandos Principales

Sprites
// Crear sprites
const player = this.add.sprite(100, 450, 'dude');

// Sprite con física
const player = this.physics.add.sprite(100, 450, 'dude');
player.setBounce(0.2);
player.setCollideWorldBounds(true);
Animaciones
// Crear animación
this.anims.create({
  key: 'left',
  frames: this.anims.generateFrameNumbers('dude', { start: 0, end: 3 }),
  frameRate: 10,
  repeat: -1
});

// Reproducir animación
player.anims.play('left');
Controles
// Crear controles
cursors = this.input.keyboard.createCursorKeys();

// En update()
if (cursors.left.isDown) {
  player.setVelocityX(-160);
} else if (cursors.right.isDown) {
  player.setVelocityX(160);
}

Física y Colisiones

Sistema de Física
// Configurar física en config
physics: {
  default: 'arcade',
  arcade: {
    gravity: { y: 300 },
    debug: false
  }
}

// Crear grupo con física
platforms = this.physics.add.staticGroup();
Colisiones
// Detectar colisiones
this.physics.add.collider(player, platforms);

// Overlap con callback
this.physics.add.overlap(player, stars, collectStar, null, this);

function collectStar(player, star) {
  star.disableBody(true, true);
}

Características Avanzadas

Escenas
// Crear clase de escena
class GameScene extends Phaser.Scene {
  constructor() {
    super({ key: 'GameScene' });
  }
}

// Cambiar escena
this.scene.start('GameScene');
Audio
// Cargar audio
this.load.audio('jump', 'assets/jump.wav');

// Reproducir sonido
const jumpSound = this.sound.add('jump');
jumpSound.play();

// Música de fondo
const music = this.sound.add('bgMusic', { loop: true });
music.play();
